# Veda runtime dependencies.
#
# Most of Veda is prose + stdlib Python (scripts/calc.py, scripts/validate_profile.py).
# This file lists the few third-party packages needed by the scripts that reach
# outside the repo.
#
# Install with:
#   python -m pip install -r requirements.txt

# scripts/fetch_quote.py, scripts/fetch_fundamentals.py - live quotes & data via yfinance.
# Pinned with a lower bound; newer versions are expected to be compatible.
# yfinance is unofficial Yahoo Finance and may rate-limit on heavy use.
yfinance>=0.2.40

# scripts/fetch_fundamentals.py - Screener.in scraping for India equities.
requests>=2.28.0

# scripts/fetch_news.py - RSS / Atom / Google News parsing for the
# news-researcher subagent. feedparser handles schema variation across feeds;
# beautifulsoup4 strips HTML out of <description> blocks.
feedparser>=6.0.10
beautifulsoup4>=4.12.0

# scripts/kite.py - Zerodha Kite Connect (holdings). Official SDK.
# Only imported inside the `auth` / `holdings` subcommands, so the rest of
# Veda runs without it installed.
kiteconnect>=4.2.0
